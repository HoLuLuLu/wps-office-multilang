本示例在活动文档中添加一条直线，然后设置其格式为带箭头的红色直线。

Dim lineNew As Shape
'Add new line to document
Set lineNew = ActiveDocument.Shapes.AddLine _
(BeginX:=100, BeginY:=100, EndX:=60, EndY:=20)
'Format line
With lineNew.Line
    .BeginArrowheadStyle = KsoArrowheadNone
    .EndArrowheadStyle = KsoArrowheadTriangle
    .ForeColor.RGB = RGB(Red:=128, Green:=0, Blue:=0)
End With
